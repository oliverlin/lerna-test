```bash
npm install --save custom-package-master
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'custom-package-master'
import 'custom-package-master/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
