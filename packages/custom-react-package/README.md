## Install

```bash
npm install --save custom-react-package
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'custom-react-package'
import 'custom-react-package/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
